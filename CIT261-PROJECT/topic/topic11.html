1.- Move the mouse over the square and wait a moment!


 div  class="originalValues" onmouseover="changeit(this)"  Move here! /div

 .originalValues{
    width: 80px; height: 20px; background-color:lightgoldenrodyellow;

     /* shorthand to say how the transition for height and width must to last */
     -webkit-transition: height 2s, width 0.5s;
     /* time before the transition start */
    -webkit-transition-delay: 1s;
     /* Same speed from start to end*/
    transition-timing-function: linear;
}

  //change the currently properties of the Div.    
 function changeit(setDiv){
    //This use all the available width.
     setDiv.style.height = "100px"; setDiv.style.width = "100%";

    //after two 2.5 seconds I display the next message.
     setTimeout(function(){ 
     setDiv.style.fontSize = "3rem";
     setDiv.style.textAlign = "center";
     setDiv.innerHTML = "Thank you for changing me!"; 
     setDiv.style.backgroundColor = "#E3CEF6";}, 2500);
}
