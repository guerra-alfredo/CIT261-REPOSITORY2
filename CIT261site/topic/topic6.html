TOPIC 06: DOM Manipulation Using createElement, appendChild, insertBefore, removeChild, etc.

    
    var h = document.createElement("H3"); // allow to create element, Heading
    var textHeading = document.createTextNode("I am a new heading - H3"); //allow to change the name of the object
    h.appendChild(textHeading); //allow to append the element with the name.
    document.body.appendChild(h); //insert the element to the body.
    
    var newItem = document.createElement("LI"); //create en li element
    var btn = document.createElement("BUTTON"); //create a button element
    var para = document.createElement("P"); //create a paragraph element
    
    // allow to insert an element at a position by the childNodes[0]
    var myNumbers = document.getElementById("numbers");
    myNumbers.insertBefore(newItem, myNumbers.childNodes[0]);
    
    
    //this code create a paragraph and save it in a DIV, by using an Id inside the DIV.

    document.getElementById("newP").appendChild(para);
